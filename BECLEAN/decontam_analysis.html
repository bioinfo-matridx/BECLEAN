run_decontam <- function(otu_mat, tax_mat, meta_mat, method, conc, normalize=FALSE) {
    rownames(meta_mat) <- c()
    samples_df <- meta_mat
    otu_mat <- otu_mat %>% tibble::column_to_rownames("taxid")
    tax_mat <- tax_mat %>% tibble::column_to_rownames("taxid")
    samples_df <- samples_df %>% tibble::column_to_rownames("sample_name")
    otu_mat <- as.matrix(otu_mat)
    otu_mat <- otu_mat[, c(rownames(samples_df))]
    tax_mat <- as.matrix(tax_mat)
    OTU = otu_table(otu_mat, taxa_are_rows = TRUE)
    TAX = tax_table(tax_mat)
    samples = sample_data(samples_df)

    data <- phyloseq(OTU, TAX, samples)
    sample_data(data)$is.neg <- sample_data(data)$sample_or_control == "control"
    results <- list()
    results$first <- isContaminant(data, method=method, neg='is.neg', conc=conc, normalize=normalize)
    results$second <- data
    return(results) 
}

taxid_label_map <- c('562'='eco', '287'='pae', '34062'='mos')
taxid_name_map <- c('562'='Escherichia coli', '287'='Pseudomonas aeruginosa', '34062'='Moraxella osloensis')
for (i in seq(1, length(taxid_name_map))) {
    to_match <- c("NC", "negative")
    colors <- c("negative" = "#00BFC4", "positive"="#F8766D")
    sp_name <- taxid_name_map[i]
    taxid <- names(sp_name)
    taxid_meta_df <- samples_df[grep(paste(to_match, collapse='|'), samples_df$sample_name),]
    res <- run_decontam(otu_mat, tax_mat, taxid_meta_df, 'combined', 'library_concentration', normalize=FALSE)
    a = plot_frequency(res$second, taxid, conc='library_concentration', normalize=FALSE) +
        xlab("Library Concentration") + ggtitle(sp_name) +
        theme(plot.title = element_text(hjust=0.5, face="bold.italic", size=18), axis.title=element_text(size=18),
             axis.text=element_text(size=16))
    a$layers = a$layers[c(1,2)]
    a = a + geom_point(aes(color=color_label), size=4, alpha=0.9) + 
        scale_color_manual(values=colors, labels = c("negative", "positive")) + guides(color=guide_legend(""))
    print(a)
}
